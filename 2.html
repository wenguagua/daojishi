    <script>

        let meetingDate = new Date();

        meetingDate.setDate(meetingDate.getDate() + 7);

        let currentAvatarType = '';


        function loadSettings() {

            const savedDate = localStorage.getItem('meetingDate');

            const partner1 = localStorage.getItem('partner1') || '我';

            const partner2 = localStorage.getItem('partner2') || '你';

            

            if (savedDate) meetingDate = new Date(savedDate);

            document.getElementById('name1').textContent = partner1;

            document.getElementById('name2').textContent = partner2;

            document.getElementById('meetingDate').value = meetingDate.toISOString().slice(0, 16);

            

            loadAvatar('avatar1');

            loadAvatar('avatar2');

        }


        function loadAvatar(avatarId) {

            const imgData = localStorage.getItem(avatarId);

            if (imgData) {

                const img = document.getElementById(avatarId);

                img.src = imgData;

                img.style.display = 'block';

                document.getElementById(avatarId + '-placeholder').style.display = 'none';

            }

        }


        function uploadAvatar(type) {

            currentAvatarType = type;

            document.getElementById('fileInput').click();

        }


        document.getElementById('fileInput').addEventListener('change', function(e) {

            const file = e.target.files[0];

            if (file && file.type.startsWith('image/')) {

                const reader = new FileReader();

                reader.onload = function(e) {

                    localStorage.setItem(currentAvatarType, e.target.result);

                    loadAvatar(currentAvatarType);

                };

                reader.readAsDataURL(file);

            }

        });


        function updateCountdown() {

            const now = new Date().getTime();

            const distance = meetingDate - now;


            if (distance < 0) {

                document.getElementById('days').textContent = '00';

                document.getElementById('hours').textContent = '00';

                document.getElementById('minutes').textContent = '00';

                document.getElementById('seconds').textContent = '00';

                document.querySelector('.poem').innerHTML = '此刻即永恒<br>我们终于抵达';

                return;

            }


            const days = Math.floor(distance / (1000 * 60 * 60 * 24));

            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

            const seconds = Math.floor((distance % (1000 * 60)) / 1000);


            document.getElementById('days').textContent = String(days).padStart(2, '0');

            document.getElementById('hours').textContent = String(hours).padStart(2, '0');

            document.getElementById('minutes').textContent = String(minutes).padStart(2, '0');

            document.getElementById('seconds').textContent = String(seconds).padStart(2, '0');

        }


        function openSettings() {

            document.getElementById('settingsModal').style.display = 'flex';

        }


        function closeSettings() {

            document.getElementById('settingsModal').style.display = 'none';

        }


        function saveSettings() {

            const dateInput = document.getElementById('meetingDate').value;

            const partner1 = document.getElementById('partner1').value || '我';

            const partner2 = document.getElementById('partner2').value || '你';

            

            if (dateInput) {

                meetingDate = new Date(dateInput);

                localStorage.setItem('meetingDate', meetingDate);

                localStorage.setItem('partner1', partner1);

                localStorage.setItem('partner2', partner2);

                loadSettings();

                closeSettings();

                updateCountdown();

            }

        }


        loadSettings();

        updateCountdown();

        setInterval(updateCountdown, 1000);

        document.getElementById('settingsModal').addEventListener('click', function(e) {

            if (e.target === this) closeSettings();

        });
